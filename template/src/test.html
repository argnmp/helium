#include <iostream>
#include <algorithm>
#include <climits>
#include <string>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
string n;
int nn, k, ans = INT_MIN;
int memo[1000001][11] = {0,};
struct box{
    string n;
    int dist;
};

void solve(){
    cin >> n >> k;
    nn = n.length();
    queue<box> q;
    q.push({n, 0});
    while(!q.empty()){
        box t = q.front();
        q.pop();
        if(t.dist==k){
            ans = max(ans, stoi(t.n)); 
            continue;
        }
        for(int i = 0; i<nn-1; i++){
            for(int j = i+1; j<nn; j++){
                if(!(i==0 && t.n[j]=='0')){
                    swap(t.n[i], t.n[j]);
                    box nt = {t.n, t.dist + 1};
                    if(memo[stoi(nt.n)][nt.dist]==0){
                        memo[stoi(nt.n)][nt.dist] = 1;
                        q.push(nt);
                    }
                    swap(t.n[i], t.n[j]);
                }             
            }
        } 
    }
    if(ans==INT_MIN)
        cout << -1;
    else
        cout << ans;
    
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 언뜻봐서는 6C2 ^ 10 으로 시간내에 전체 탐색이 불가능해 보인다. 하지만, 실제로는 중복된 경우가 많이 발생하기 때문에 충분히 시간내에 해결이 가능하다.
  • 중복된 것의 의미는 다음과 같다
    • 12345 라는 숫자가 있을 때, 2번 교환을 한다고 해보자
    • 1과 2를 2번 교환한다면 12345가 나온다
    • 1과 3을 2번 교환한다면 12345가 나온다.
    • ... 이런식으로 2번 교환했을 때 12345가 매우 많이 중복이 된다.
  • 따라서 중복되는 경우를 한번씩만 계산하도록 2차원 배열로 memo를 해두고, 1번 교환 부터 K번 교환까지 점점 늘려가면서 BFS를 해주면 된다.
